Bootstrap Prompts

Bootstrap modals for alert() and confirm().

alert()

alert("Push the button.");

try it! 

alert("a message", "a title");

try it! 

alert.original("Oldschool.");

try it!

 

confirm()

confirm("Ok?", function(result) {
    console.log(result);
});

try it! 

confirm("Ok?", "Title", function(result) {
    console.log(result);
});

try it! 

confirm.original("Ok?", function(result) {
    console.log(result);
});

try it!
